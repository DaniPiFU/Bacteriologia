    <p id="resultado"></p>

    <script>
        function identificarBacteria() {
            const bacterias = [
                { nombre: "Escherichia coli", forma: "Bacilo", gram: "Gram negativa", movilidad: "Móvil" },
                { nombre: "Staphylococcus aureus", forma: "Coco", gram: "Gram positiva", movilidad: "No móvil" },
                { nombre: "Bacillus subtilis", forma: "Bacilo", gram: "Gram positiva", movilidad: "Móvil" },
                { nombre: "Streptococcus pneumoniae", forma: "Coco", gram: "Gram positiva", movilidad: "No móvil" },
            ];

            const forma = document.getElementById("forma").value;
            const gram = document.getElementById("gram").value;
            const movilidad = document.getElementById("movilidad").value;

            const bacteria = bacterias.find(b => b.forma === forma && b.gram === gram && b.movilidad === movilidad);

            document.getElementById("resultado").textContent = bacteria ? La bacteria identificada es: ${bacteria.nombre} : "Bacteria no encontrada";
        }
    </script>
